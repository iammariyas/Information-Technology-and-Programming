# Частотный анализ на минималках

```
s = []
while (frase := input()) != 'ФИНИШ':
    s.append(frase.lower())
s = ''.join(s)
maximum = 0
periodicity = ''
for i in s:
    cnt = s.count(i)
    if cnt > maximum:
        maximum = cnt
        periodicity = i
    elif cnt == maximum:
        if i < periodicity:
            periodicity = i
print(periodicity)
```

n = input()
s = n.split()
a = [int(x) for x in s]
num1 = a[0]
while len(a) > 1:
    num2 = a.pop(1)
    while num1 != num2:
        if num1 > num2:
            num1 -= num2
        else:
            num2 -= num1
print(num1)

    if s[i] == s[i + 1]:
        cnt += 1
    else:
