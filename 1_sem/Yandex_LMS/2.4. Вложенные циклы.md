# Максимальная сумма

```
n = int(input())
maximum = 0
winner = ''
while n != 0:
    name = input()
    num = int(input())
    cnt = 0
    while num > 0:
        cnt += num % 10
        num //= 10
    if cnt >= maximum:
        maximum = cnt
        winner = name
    n -= 1
print(winner)
```

# Числовая змейка

```
n = int(input())
m = int(input())
printed_str = m
leight = len(str(n * m))
for h in range(n):
    if h % 2 == 0:
        for w in range(h * m + 1, printed_str + 1):
            print(f'{w:>{leight}}', end=' ')
    else:
        for w in range(printed_str, h * m, -1):
            print(f'{w:>{leight}}', end=' ')
    printed_str += m
    print()
```

# Числовая змейка 2.0

```
n = int(input())
m = int(input())
printed_str = 0
leight = len(str(n * m))
for h in range(n):
    for w in range(m):
        if w % 2 == 0:
            printed_str = w * n + h + 1
        else:
            printed_str = (w + 1) * n - h
        print(f'{printed_str:>{leight}}', end=' ')
    print()
```

# Редизайн таблицы умножения

```
size = int(input())
weight = int(input())
cnt = 0
lenght = weight * size + (size - 1)
for x in range(1, size + 1):
    for y in range(1, size + 1):
        if y != size:
            print(f'{x * y:^{weight}}', end='|')
    else:
        print(f'{x * y:^{weight}}')
    if x != size:
        print('-' * lenght)
    else:
        print()
```

# Математическая выгода

```
n = int(input())
max_base = 0
max_sum = 0
for base in range(10, 1, -1):
    s = 0
    num = n
    while num != 0:
        s += (num % base)
        num //= base
    if s >= max_sum:
        max_sum = s
        max_base = base
print(max_base)
```
