# Корень зла

```
a = float(input())
b = float(input())
c = float(input())
d = b ** 2 - 4 * a * c
if a == b == c == 0:
    print('Infinite solutions')
elif a == b == 0:
    print('No solution')
else:
    if a == 0:
        print(f'{(-c / b):.2f}')
    if b == 0:
        if c > 0:
            x1 = (c / a) ** 0.5
            x2 = -((c / a) ** 0.5)
            print(f'{min(x1, x2):.2f} {max(x1, x2):.2f}')
        else:
            print('No solution')
if a != 0:
    if b == c == 0:
        print(f'{(0 / a):.2f}')
    if d < 0:
        print('No solution')
    if d == 0:
        print(f'{-b / (2 * a):.2f}')
    if d > 0:
        x1 = (-b + d ** 0.5) / (2 * a)
        x2 = (-b - d ** 0.5) / (2 * a)
        print(f'{min(x1, x2):.2f} {max(x1, x2):.2f}')
```
