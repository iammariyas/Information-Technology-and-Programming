# Маршрут построен

```
coordinate_x = 0
coordinate_y = 0
while (direction := input()) != 'СТОП':
    steps = int(input())
    if direction == 'СЕВЕР':
        coordinate_y += steps
    elif direction == 'ЮГ':
        coordinate_y -= steps
    elif direction == 'ВОСТОК':
        coordinate_x += steps
    elif direction == 'ЗАПАД':
        coordinate_x -= steps
print(coordinate_y)
print(coordinate_x)
```

# Чётная чистота

```
n = int(input())
cnt = 1
res = 0
while n != 0:
    s = n % 10
    if s % 2 != 0:
        res += s * cnt
        cnt *= 10
    n //= 10
print(res)
```

# Простая задача 2.0

```
n = int(input())
d = 2
res = []
while n > 1:
    if n % d == 0:
        n //= d
        res.append(d)
    else:
        d += 1
print(' * '.join(str(x) for x in res))
```

# Игра в «Угадайку»

```

b = 1000
c = (a + b) // 2
print(c)
while (frase := input()) != 'Угадал!':
    if frase == 'Больше':
        a = c + 1
    elif frase == 'Меньше':
        b = c - 1
    c = (a + b) // 2
    print(c)
```

# Хайпанём немножечко!
