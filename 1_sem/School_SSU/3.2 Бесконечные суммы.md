# Вложенные циклы. Задача 8.2

```
h, eps = map(float, input().split())
x = 0.1
while x <= 0.7 + eps:
    p1 = 3
    z1 = 2
    y = x / 2
    number_series = 0
    while y > eps:
        number_series += y
        y = x ** p1 / 2 ** z1
        p1 += 2
        z1 += 1
    print(f'{x:.8f}', f'{number_series:.8f}')
    x += h
```

# Вложенные циклы. Задача 8.7

```
h, eps = map(float, input().split())
x = 0.1
while x < 3.0 + 1e-8:
    y0 = 0.0
    y1 = 0.5
    number_series = y0
    i = 1
    f = True
    while abs(y1) > eps:
        number_series += y1
        y0, y1 = y1, (y0 + y1) / i + (y0 ** 3 * x ** 2) / i
        i += 1
        if i > 10000 or number_series > 10 ** 9:
            f = False
            break
    if f:
        print(f'{x:.8f} {number_series:.8f}')
    else:
        print(f'{x:.8f} 0.00000000')
    x += h
```
