# Таблица значений функции 1

```
a, b, h = map(float, input().split())
x = a
while not (x - b > 1e-6):
    if abs(x + 1) < 1e-6:
        print(f'{x:.6f} undefined')
    else:
        print(f'{x:.6f} {(1 / (1 + x) ** 2):.6f}')
    x += h
```

# Таблица значений функции 6

```
from math import log

a, b, h = map(float, input().split())
x = a
while x - b < 1e-9:
    if 4 - x ** 2 < 1e-9:
        print(f'{x:.6f} undefined')
    else:
        print(f'{x:.6f} {log(4 - x ** 2):.6f}')
    x = float(f'{x + h:.6f}')
```

# Таблица значений функции 10

```
from math import log
a, b, h = map(float, input().split())
x = a
while x - b <= 1e-6:
    if abs(x - 2) < 1e-6:
        print(f'{x:.6f} undefined')
    else:
        print(f'{x:.6f} {log(abs(x - 2)):.6f}')
    x = float(f'{x + h:.6f}')
```
