# ЕГЭ. Задача 16

```
from collections import Counter

s = []
while True:
    line = input().strip().lower()
    s.append(line)
    if '.' in line:
        break

text = ''.join(s)

symbol = Counter(char for char in text if char.isalpha())

max_value = max(symbol.values())
max_letter = [key for key, value in symbol.items() if value == max_value]
print(min(max_letter), max_value)
```
# Две строки 3

```
from collections import Counter

s1 = Counter(input())
s2 = Counter(input().strip())
print(all(s2[letter] <= s1.get(letter, 0) for letter in s2))
```
# Расшифровка оценок

```
s = input()
grade = {2: 0, 3: 0, 4: 0, 5: 0}
i = 0
while i < len(s):
    if s[i] == '0':
        grade[2] += 1
        i += 1
    elif s[i:i + 3] == '100':
        grade[3] += 1
        i += 3
    elif s[i:i + 3] == '101':
        grade[4] += 1
        i += 3
    elif s[i:i + 2] == '11':
        grade[5] += 1
        i += 2
    else:
        i += 1

max_grade = max(grade, key=grade.get)
print(max_grade)
```
