# Рекурсивный сумматор

```
def recursive_sum(*args):
    if args:
        return args[0] + recursive_sum(*args[1:])
    else:
        return 0
```

# Рекурсивный сумматор цифр

```
def recursive_digit_sum(num):
    if num != 0:
        return num % 10 + recursive_digit_sum(num // 10)
    else:
        return 0
```

def make_equation(*args):
    if len(args) > 1:
        return f'({make_equation(*args[:-1])}) * x + {str(args[-1])}'
    else:
        return str(args[0])

# Сортировка слиянием

```
def merge(left, right):
    sort_list = []
    index_left = index_right = 0

    for _ in range(len(left) + len(right)):
        if index_left < len(left) and index_right < len(right):
            if left[index_left] <= right[index_right]:
                sort_list.append(left[index_left])
                index_left += 1
            else:
                sort_list.append(right[index_right])
                index_right += 1
        elif index_left == len(left):
            sort_list.append(right[index_right])
            index_right += 1
        elif index_right == len(right):
            sort_list.append(left[index_left])
            index_left += 1
    return sort_list


def merge_sort(nums):
    if len(nums) <= 1:
        return nums
    median = len(nums) // 2
    left = merge_sort(nums[:median])
    right = merge_sort(nums[median:])
    return merge(left, right)
```

# "Выпрямление" списка

```
def make_linear(x):
    if not x:
        return x
    if isinstance(x[0], list):
        return make_linear(x[0]) + make_linear(x[1:])
    return x[:1] + make_linear(x[1:])
```
