# Разделяй и властвуй

```
file, evens_file, odds_file, equals_file = input(), input(), input(), input()

with open(file, encoding='utf-8') as numbers:
    num = [x for x in numbers.read().split('\n') if x]

even_digit = '02468'
odds_digit = '13579'

for s in num:
    evens, odds, equals = [], [], []
    for number in s.split():
        all_evens = all_odds = 0
        for x in number:
            if x in even_digit:
                all_evens += 1
            elif x in odds_digit:
                all_odds += 1

        if all_odds > all_evens:
            odds.append(number)
        elif all_odds < all_evens:
            evens.append(number)
        else:
            equals.append(number)

    with open(odds_file, 'a', encoding='utf-8') as f1:
        f1.write(f'{' '.join(odds)}\n')
    with open(evens_file, 'a', encoding='utf-8') as f2:
        f2.write(f'{' '.join(evens)}\n')
    with open(equals_file, 'a', encoding='utf-8') as f3:
        f3.write(f'{' '.join(equals)}\n')
```

# Слияние данных

```
import json

file, n = input(), input()
with open(file, encoding='utf-8') as f:
    users = json.load(f)
with open(n, encoding='utf-8') as us:
    upd = json.load(us)

for update in upd:
    for user in users:
        if update['name'] == user['name']:
            for key in update.keys():
                if update[key] > user.get(key, ''):
                    user[key] = update[key]

res = {}
for user in users:
    name = user.pop('name')
    res[name] = user

with open(file, 'w', encoding='utf-8') as f:
    json.dump(res, f, ensure_ascii=False, indent=4)
```
