# Однофамильцы

```
n = int(input())
res = {}
a = []
for i in range(n):
    surname = input().split()
    for name in surname:
        res.setdefault(name, 0)
        res[name] += 1
for value in res.values():
    if value > 1:
        a.append(value)
print(sum(a))
```

```
    surname = input()

        res[surname] = 0
    res[surname] += 1
res = {key: value for key, value in sorted(res.items()) if value > 1}
if res:
    for key, value in res.items():
        print(f'{key} - {value}')
```

# Это будет шедевр!

```
n = int(input())
food = set()
for _ in range(n):
    food.add(input())
res = set()
m = int(input())
for i in range(m):
    name = input()
    ingredients = int(input())
    receipt = set()
    for j in range(ingredients):
        receipt.add(input())
    if len(receipt - food) == 0:
        res.add(name)

if len(res) > 0:
    print('\n'.join(sorted(res)))
else:
    print('Готовить нечего')
```

# Зайка — 10

```
res = set()
while frase := input().split():
    for i, el in enumerate(frase):
        if el == 'зайка' and i not in (0, len(frase) - 1):
            res.add(frase[i - 1])
            res.add(frase[i + 1])
        elif el == 'зайка' and not i:
            res.add(frase[i + 1])
        elif el == 'зайка' and i == len(frase) - 1:
            res.add(frase[i - 1])

for el in res:
    print(el)
```
