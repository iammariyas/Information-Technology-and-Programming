# Генератор матриц

```
def make_matrix(size, value=0) -> list:
    if isinstance(size, int):
        return [[value for i in range(size)] for i in range(size)]
    return [[value for i in range(size[0])] for i in range(size[1])]
```

# Функциональный нод 2.0

```
def gcd(*args) -> int:
    num = list(args)
    while len(num) > 1:
        while num[1]:
            num[0], num[1] = num[1], num[0] % num[1]
        num.pop(1)
    return num[0]
```

def month(num, language='ru'):
    month_name = {
        'ru': ['Январь', 'Февраль', 'Март', 'Апрель', 'Май', 'Июнь',

nums = []


def enter_results(*args):
    nums.extend(args)


def get_sum():
    first_sum = sum(nums[::2])
    second_sum = sum(nums[1::2])
    return round(first_sum, 2), round(second_sum, 2)


def get_average():
    first_num = nums[::2]
    second_num = nums[1::2]
    first_average = sum(first_num) / len(first_num)
    second_average = sum(second_num) / len(second_num)
    return round(first_average, 2), round(second_average, 2)

```
def secret_replace(text, **replaces):
    res = ''
    symbols = {key: (value, 0) for key, value in replaces.items()}
    for symbol in text:
        if symbol in symbols:
            pos = symbols[symbol][1] % len(symbols[symbol][0])
            res += symbols[symbol][0][pos]
            symbols[symbol] = symbols[symbol][0], symbols[symbol][1] + 1
        else:
            res += symbol

    return res
```
